# mylife V1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Mac-first local task manager with today-first tri-pane UI, task trees, Markdown checkbox sync, reminders, gantt timeline, and always-on-top desktop widget.

**Architecture:** Use a Tauri desktop shell with a React frontend and a Go local API process backed by SQLite. The frontend owns interaction and rendering, while the Go core owns task state transitions, event history, reminder scheduling, and Markdown checkbox-to-task synchronization. All reads and writes go to the local database so the app works offline.

**Tech Stack:** Tauri, React, TypeScript, Vite, Vitest, Playwright, Go, Gin, modernc.org/sqlite, testify

func (h *TaskHandler) CreateTask(c *gin.Context) {
	var in domain.Task
	if err := c.ShouldBindJSON(&in); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
		return
	}
	task, err := h.repo.Create(c.Request.Context(), in)
	if err != nil {
		c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
		return
	}
	c.JSON(http.StatusCreated, task)
}

func (h *TaskHandler) ListTasks(c *gin.Context) {
	status := c.Query("status")
	today := c.Query("today") == "true"
	tasks, err := h.repo.List(c.Request.Context(), status, today)
	if err != nil {
		c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
		return
	}
	c.JSON(http.StatusOK, gin.H{"items": tasks})
}

// apps/core/internal/service/task_service.go
func (s *TaskService) TransitionTask(ctx context.Context, taskID string, toStatus string) (*TransitionResult, error) {
	task, err := s.repo.Get(ctx, taskID)
	if err != nil {
		return nil, err
	}
	eventID := newID()
	task.Status = toStatus
	task.Color = domain.DefaultStatusColors[toStatus]
	if err := s.repo.UpdateStatus(ctx, task, domain.TaskEvent{
		ID:         eventID,
		TaskID:     task.ID,
		FromStatus: task.Status,
		ToStatus:   toStatus,
		EventType:  "status_changed",
	}); err != nil {
		return nil, err
	}
	return &TransitionResult{Task: *task, UndoToken: eventID}, nil
}

func (s *TaskService) UndoTransition(ctx context.Context, taskID string, undoToken string) (*domain.Task, error) {
	return s.repo.UndoLastTransition(ctx, taskID, undoToken)
}

// apps/core/internal/markdown/parser.go
type CheckboxTask struct {
	Title  string
	Status string
}

func ParseCheckboxTasks(input string) []CheckboxTask {
	lines := strings.Split(input, "\n")
	items := make([]CheckboxTask, 0)
	for _, line := range lines {
		switch {
		case strings.HasPrefix(line, "- [ ] "):
			items = append(items, CheckboxTask{Title: strings.TrimPrefix(line, "- [ ] "), Status: "not_started"})
		case strings.HasPrefix(line, "- [x] "):
			items = append(items, CheckboxTask{Title: strings.TrimPrefix(line, "- [x] "), Status: "completed"})
		}
	}
	return items
}

// apps/core/internal/reminder/scheduler.go
type Rule struct {
	Kind     string
	Weekdays []time.Weekday
}

func NextOccurrence(rule Rule, from time.Time) time.Time {
	if rule.Kind == "weekly" {
		for i := 1; i <= 7; i++ {
			next := from.AddDate(0, 0, i)
			for _, weekday := range rule.Weekdays {
				if next.Weekday() == weekday {
					return next
